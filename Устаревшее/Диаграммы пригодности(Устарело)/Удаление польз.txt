control "Закрытие окна" as co3

control "Добавление данных" as co4

boundary "Домашнее окно администратора" as bo

boundary "Окно удаления пользователя" as bo1

bo1 --> co3 : Нажата кнопка\n Отмена

co4 --> en
co2 --> err
err -- pl
