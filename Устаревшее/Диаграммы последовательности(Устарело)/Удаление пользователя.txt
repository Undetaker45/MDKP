@startuml

actor "Администратор" as pl

boundary "Домашнее окно\n администратора" as bo

control "Открытие окна\n удаления пользователя" as co

boundary "Окно удаления\n пользователя" as bo1

control "Проверка данных" as co2

control "Закрытие окна" as co3

control "Добавление данных" as co4

entity "Файловая система" as en

boundary "Окно ошибки" as err

pl -> bo

bo -> co : Нажата кнопка\n Удаление пользователя

co -> bo1

bo1 -> co3 : Нажата кнопка\n Отмена

bo1 -> co2 : Нажата кнопка\n Подтвердить

co2 -> en : Запрос

en --> co2 : Ответ

co2 --> co2 : Проверка ответа

alt Данные корректны

co2 -> co4 : Данные корректны

else Данные не корректны

co4 -> en

co2 -> err : Данные не коректны

err -> pl

end

@enduml
